// Preprocessor directive
#include <iostream>

// Standard namespace
using namespace std;

// Instantiate the main function
int main()
{

// Declare a char to hold user input
char age;

// Displays message asking for user input.
cout << "Prices on our buffet varies by age. Please select your age range.\n" <<

//Choice 1
"1. Under 12 Years Old\n" <<

//Choice 2
"2. 13 Years And Older\n" <<

//Choice 3
"3. 18 Years And Older\n" <<

//Choice 4
"4. 60 Years And Older\n";

//Saves user input to age variable
cin >> age;

//Display Message
cout << "You are a(n) ";

//Switch Statement
switch (age) {

//Display Message if Selected Choice 1
case '1':
cout << "Child. The price will be $5.99" << endl;
break;

//Display Message if Selected Choice 2
case '2':
cout << "Teenager. The price will be $14.99" << endl;
break;

//Display Message if Selected Choice 3
case '3':
cout << "Adult. The price will be $12.99" << endl;
break;

//Display Message if Selected Choice 4
case '4':
cout << "Senior. The price will be $8.99" << endl;
break;

//Display message if selected choice is not listed
default:
cout << "This is not a valid option." << endl;
break;

//Exit switch conditional
}

// Return status 0 from main function
return 0;
}
